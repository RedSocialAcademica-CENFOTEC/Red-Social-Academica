cmake_minimum_required(VERSION 3.16)
project(RedSocialAcademicaGUI LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Qt necesita que estas 3 opciones esten prendidas para generar moc/uic/rcc automaticamente
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

find_package(Qt6 COMPONENTS Widgets REQUIRED)

add_executable(red_social_gui
    main_gui.cpp
    mainwindow.h
    mainwindow.cpp
    grafo.h
    grafo.cpp
    gestorusuarios.h
    gestorusuarios.cpp
    gestorsolicitudes.h
    gestorsolicitudes.cpp
    gestorcomentarios.h
    gestorcomentarios.cpp
    bplustree.h
    bplustree.cpp
    dynamichash.h
    dynamichash.cpp
    notificationqueue.h
    notificationqueue.cpp
    modelos.h
)

target_link_libraries(red_social_gui PRIVATE Qt6::Widgets)
